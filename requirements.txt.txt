from flask import Flask, jsonify
from src.data_loader import load_csv
from src.preprocess import preprocess_telco, get_feature_types
from src.imputers import impute_median_mode, missing_summary
from src.evaluate import dataset_summary

app = Flask(__name__)


@app.route("/", methods=["GET"])
def home():
    return jsonify({
        "project": "Imputation Pipeline API for Churn Data",
        "status": "running",
        "message": "API is ready."
    })


@app.route("/demo", methods=["GET"])
def demo():
    """
    Demo route using the Telco dataset if present locally.
    """
    try:
        df = load_csv("data/Telco-Customer-Churn.csv")
        df = preprocess_telco(df)

        num_cols, cat_cols = get_feature_types(df)

        before = dataset_summary(df)
        missing_before = missing_summary(df)

        df_imputed = impute_median_mode(df, num_cols, cat_cols)

        after = dataset_summary(df_imputed)
        missing_after = missing_summary(df_imputed)

        return jsonify({
            "before": before,
            "after": after,
            "missing_before": missing_before,
            "missing_after": missing_after
        })
    except Exception as e:
        return jsonify({"error": str(e)}), 500


if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000, debug=True)
